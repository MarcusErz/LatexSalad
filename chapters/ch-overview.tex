\chapter{Overview}

\section{Struktur des Papers}

\begin{enumerate}
	\item Titel
	\item Authors
	\item Keywords
	\item Abstract (nicht mehr als 300 Wörter)
	\item Introduction (Hier soll auch sowas wie related work rein)
	\item Method (Was wurde gemacht. Andere Forscher sollen dazu befähigt werden das ganz nachzubauen)
	\item Results
	\item Discussion (Die Results sollten hierbei diskutiert werden und mit anderen Ergebnissen verglichen werden)
	\item Conclusion (Die Neuheit und Wichtigkeit hervorheben)
	\item Acknowledgments (Denke brauchen wir nicht)
	\item References
\end{enumerate}
